    for field in jsondata['document']:
        try:
            key = field['name']
            value = field['value']

            if key in mandatories:
                mandatories.remove(key)
            else:
                optionals.remove(key)
        except (ValueError, KeyError):
            return False

        try:
            function_dict[key](value)
